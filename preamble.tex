% Set up command for Yaku
\def\rulesetprefix{undef}
\newcommand{\yaku}[5]{
	\def\e{}\def\t{#4}
	% #1 - Romaji Name
	% #2 - English Name
	% #3 - Japanese Name
	% #4 - Flags for the Yaku
	% #5 - Description of the Yaku
	\subsubsection*{\large #1~--~{\cjkbf #3} {\normalfont\itshape\normalsize``#2''\/}}\label{\rulesetprefix:yaku:#1}
	\vspace*{-12pt}
	\ifx\e\t\else \textit{\footnotesize #4\\} \fi
	\hspace*{\parindent}#5
}
